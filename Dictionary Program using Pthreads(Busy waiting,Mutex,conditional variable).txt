#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <pthread.h>

#define BUFFER_SIZE 10
#define THREADS 3
#define WORD_LEN 20

const char *dictionary[] = {
    "mango","education","kite","eagle","naughty",
    "wall","doll","book","parrot","god"
};

int dict_size = 10;


const char *input_words[] = {
    "note","book","education","eagle","watch"
};

int total_words = 5;


char work_buffer[BUFFER_SIZE][WORD_LEN];
int work_count = 0;
int next_word = 0;


char result_buffer[BUFFER_SIZE][50];
int result_count = 0;

pthread_mutex_t mutex;
pthread_cond_t cond;

int check_dictionary(char word[])
{
    for(int i = 0; i < dict_size; i++)
    {
        if(strcmp(word, dictionary[i]) == 0)
            return 1;
    }
    return 0;
}


void *spell_check(void *arg)
{
    while(1)
    {
        pthread_mutex_lock(&mutex);

        while(work_count == 0 && next_word < total_words)
            pthread_cond_wait(&cond, &mutex);

        if(work_count == 0 && next_word >= total_words)
        {
            pthread_mutex_unlock(&mutex);
            break;
        }

        char word[WORD_LEN];
        strcpy(word, work_buffer[--work_count]);

        pthread_mutex_unlock(&mutex);

        int correct = check_dictionary(word);

        pthread_mutex_lock(&mutex);

        if(correct)
            sprintf(result_buffer[result_count++], "%s : Spell Correct", word);
        else
            sprintf(result_buffer[result_count++], "%s : Spell Incorrect", word);

        pthread_mutex_unlock(&mutex);
    }

    return NULL;
}

int main()
{
    pthread_t threads[THREADS];

    pthread_mutex_init(&mutex,NULL);
    pthread_cond_init(&cond,NULL);


    for(int i=0;i<THREADS;i++)
        pthread_create(&threads[i],NULL,spell_check,NULL);


    while(next_word < total_words)
    {
        pthread_mutex_lock(&mutex);

        strcpy(work_buffer[work_count++], input_words[next_word++]);

        pthread_cond_signal(&cond);

        pthread_mutex_unlock(&mutex);
    }

    pthread_cond_broadcast(&cond);
    for(int i=0;i<THREADS;i++)
        pthread_join(threads[i],NULL);

    printf("\nSpell Check Results:\n");

    for(int i=0;i<result_count;i++)
        printf("%s\n",result_buffer[i]);

    return 0;
}
